Function uçhariçort(alan As Range, Uç As Variant)
Dim aratoplam As Double
Dim enbüyükler As Double
Dim enküçükler As Double
 
For I = 1 To Uç
    enbüyükler = enbüyükler + WorksheetFunction.Large(alan, I)
Next I
   
For I = 1 To Uç
    enküçkler = enküçkler + WorksheetFunction.Small(alan, I)
Next I
    
aratoplam = WorksheetFunction.Sum(alan) - enbüyükler - enküçkler
uçhariçort = aratoplam / (alan.Count - Uç * 2)
 
End Function

Function STATFUNCTION(rng, op)
    Select Case UCase(op)
        Case "SUM"
            STATFUNCTION = Application.Sum(rng)
        Case "AVERAGE"
            STATFUNCTION = Application.Average(rng)
        Case "MEDIAN"
            STATFUNCTION = Application.Median(rng)
        Case "MODE"
            STATFUNCTION = Application.Mode(rng)
        Case "COUNT"
            STATFUNCTION = Application.Count(rng)
        Case "MAX"
            STATFUNCTION = Application.Max(rng)
        Case "MIN"

        Case "VAR"

        Case "STDEV"

        Case Else
            STATFUNCTION = Evaluate("NA()")
    End Select
End Function
